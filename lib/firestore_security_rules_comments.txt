// ========== قواعد أمان Firestore المقترحة ==========
// ملف توثيق فقط — انسخ إلى `firestore.rules` واضبط حسب بيئتك قبل النشر.

// 1. مجموعة users
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null &&
        (request.auth.uid == userId ||
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'full_admin');
    }

// 2. مجموعة audit_logs
    match /audit_logs/{logId} {
      allow read: if request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'full_admin';
      allow write: if request.auth != null;
    }

// 3. مجموعة admin_notifications
    match /admin_notifications/{notificationId} {
      allow read: if request.auth != null &&
        (resource.data.targetRole == '__all__' ||
         resource.data.targetRole == get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role);
      allow write: if request.auth != null;
    }
  }
}
